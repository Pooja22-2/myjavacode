package basicsjava;

public class Assignment4_AB39_Decl_Initi_Each_Datatype 
{
	static void a() 
	{
		byte a=10;
		System.out.println(a);
	}	
	static void b()
	{
		short b=32123;
		System.out.println(b);
	}	
	static void c() 
	{
		int c=98765;
		System.out.println(c);
	}	
	static void d() 
	{
		float d=5.43f;
		System.out.println(d);
	}	
	static void a1()
	{
		double a1=876.432;
		System.out.println(a1);
	}
	static void b1()
	{
		char b1='P';
		System.out.println(b1);
     }	
	static void c1() 
	{
		boolean c1=true;
		System.out.println(c1);
	}
	static void c2() 
	{
		boolean c2=false;
		System.out.println(c2);
	}
	static void name() 
	{
		String name="Pooja Jairam";
		System.out.println(name);
		
	}
	
	public static void main(String[] args) 
	{
		a();
		b();
		c();
		d();
		a1();
		b1();
		c1();
		c2();
		name();
	}
	}
